### 3.2 ==需求概述==

#### 3.2.1 目标

本系统的主要目标是：
1. 提供一个集成化的实验教学平台，解决传统实验教学中的痛点：
   - 实验过程难以追踪
   - 评分标准不统一
   - 师生互动效率低
   - 实验资源管理分散

2. 主要功能流程：
   - 教师创建实验 -> 发布实验 -> 学生执行实验 -> 自动评测 -> 成绩反馈
   - 支持实验全生命周期管理
   - 提供实时答疑和交流功能

3. 系统架构图：
```
+----------------+     +----------------+     +----------------+
|   用户界面层    |     |    业务逻辑层   |     |    数据存储层   |
+----------------+     +----------------+     +----------------+
| - Web前端界面   |     | - 实验管理模块  |     | - 用户数据     |
| - 移动端适配    |     | - 评测系统     |     | - 实验数据     |
| - 统一认证     |     | - 权限管理     |     | - 评测记录     |
+----------------+     +----------------+     +----------------+
```

```typescript
async function publishExperimentWorkflow(expId: string, config: PublishConfig) {
    // 1. 权限检查
    await userService.checkPermission(currentUser, 'PUBLISH_EXPERIMENT');
    
    // 2. 获取实验信息
    const experiment = await experimentService.getExperiment(expId);
    
    // 3. 验证实验配置
    validateExperimentConfig(experiment);
    
    // 4. 创建学生实验实例
    await experimentService.createStudentInstances(expId, config.studentIds);
    
    // 5. 发送通知
    await notificationService.sendBatchNotifications({
        type: 'EXPERIMENT_PUBLISHED',
        targetUsers: config.studentIds,
        content: generateNotificationContent(experiment)
    });
}
```

2. 实验提交评测流程：
```typescript
async function submissionWorkflow(submission: SubmissionData) {
    // 1. 保存提交
    const submissionId = await submissionService.save(submission);
    
    // 2. 触发自动评测
    const gradingTask = await gradingService.createTask(submissionId);
    
    // 3. 执行评测
    const result = await gradingService.executeTask(gradingTask);
    
    // 4. 更新成绩
    await gradeService.updateGrade(submission.userId, submission.expId, result);
    
    // 5. 发送评测结果通知
    await notificationService.sendNotification({
        userId: submission.userId,
        type: 'GRADING_COMPLETE',
        content: generateGradingNotification(result)
    });
}
```
